	\mathbb{E}[max\{X^2, Y^2\}] &= \mathbb{E}[\frac{1}{2}(X^2 + Y^2) + \frac{1}{2}\mathbb{E}[~|X^2 - Y^2|~] \\

	&= 1 + \frac{\mathbb{E}[~|X^2 - Y^2|~]}{2} \\
	&\le 2 \le 1 + \sqrt{1 - p^2}
\end{align*}
